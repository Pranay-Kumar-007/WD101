```python
from selenium import webdriver
from selenium.webdriver.firefox.service import Service
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from selenium.common.exceptions import TimeoutException, NoSuchElementException
import time

# Initialize Firefox driver
driver = webdriver.Firefox(service=Service("C:\\WebDrivers\\geckodriver.exe"))
driver.maximize_window()

try:
    # Navigate to the URL
    print("Loading page...")
    driver.get("https://developer.cisco.com/meraki/api-v1/create-organization-action-batch/")
    wait = WebDriverWait(driver, 30)
    time.sleep(3)  # Wait for page to fully load

    # Wait for and switch to the API documentation iframe
    print("Switching to iframe...")
    iframe = wait.until(EC.presence_of_element_located((By.CSS_SELECTOR, "iframe")))
    driver.switch_to.frame(iframe)
    time.sleep(2)

    # Zoom out to 50% to make more content visible
    print("Zooming out to 50%...")
    driver.execute_script("document.body.style.zoom='50%'")
    time.sleep(2)  # Wait for zoom to apply

    # Find the main scrollable content area
    print("Finding scrollable content area...")
    scroll_container = None
    
    selectors = [
        "div.api-content",
        "div[class*='ps']",
        "div.middle-panel",
        "div[role='main']",
        "#swagger-ui"
    ]
    
    for selector in selectors:
        try:
            scroll_container = driver.find_element(By.CSS_SELECTOR, selector)
            print(f"Found scroll container: {selector}")
            break
        except:
            continue
    
    if not scroll_container:
        scroll_container = driver.find_element(By.TAG_NAME, "body")
        print("Using body as scroll container")

    # Scroll down to find Schema Definition
    print("Scrolling to find Schema Definition...")
    schema_found = False
    max_scrolls = 100
    
    for i in range(max_scrolls):
        # Try to find Schema Definition button/heading
        try:
            schema_element = driver.find_element(
                By.XPATH, 
                "//button[contains(text(), 'Schema Definition')] | //h3[contains(text(), 'Schema Definition')] | //h4[contains(text(), 'Schema Definition')] | //*[contains(text(), 'Schema Definition')]"
            )
            
            if schema_element.is_displayed():
                print(f"Found Schema Definition after {i} scroll attempts")
                schema_found = True
                
                # Scroll element into view
                driver.execute_script(
                    "arguments[0].scrollIntoView({behavior: 'smooth', block: 'center'});", 
                    schema_element
                )
                time.sleep(1)
                break
                
        except NoSuchElementException:
            pass
        
        # Scroll using JavaScript
        driver.execute_script("window.scrollBy(0, 500);")
        time.sleep(0.2)
        
        # Also scroll the container if found
        if scroll_container:
            driver.execute_script("arguments[0].scrollTop += 500;", scroll_container)
        
        # Check if we've reached the bottom
        at_bottom = driver.execute_script(
            "return (window.innerHeight + window.pageYOffset) >= document.body.offsetHeight;"
        )
        if at_bottom:
            print("Reached bottom of page")
            try:
                schema_element = driver.find_element(
                    By.XPATH, 
                    "//*[contains(text(), 'Schema Definition')]"
                )
                schema_found = True
                driver.execute_script(
                    "arguments[0].scrollIntoView({behavior: 'smooth', block: 'center'});", 
                    schema_element
                )
                time.sleep(1)
                break
            except:
                pass
            break

    if not schema_found:
        print("Schema Definition not found. Taking debug screenshot...")
        driver.save_screenshot("debug_zoomed_out.png")
        raise Exception("Could not find Schema Definition section!")

    # Click to expand if it's a button/accordion
    try:
        if schema_element.tag_name == "button":
            print("Clicking Schema Definition button...")
            # Try regular click first
            try:
                schema_element.click()
            except:
                # Fallback to JavaScript click
                driver.execute_script("arguments[0].click();", schema_element)
            time.sleep(2)
    except Exception as e:
        print(f"Element might not be clickable or already expanded: {e}")

    # Find the schema content area
    print("Locating schema content...")
    schema_content = None
    
    # Try multiple ways to find the schema content
    content_selectors = [
        "//button[contains(text(), 'Schema Definition')]/following::div[contains(@class, 'accordion-body')][1]",
        "//button[contains(text(), 'Schema Definition')]/parent::*/following-sibling::div[1]",
        "//*[contains(text(), 'Schema Definition')]/following::div[contains(@class, 'model')][1]",
        "//*[contains(text(), 'Schema Definition')]/ancestor::div[contains(@class, 'opblock-section')][1]",
        "//button[contains(text(), 'Schema Definition')]/ancestor::div[contains(@class, 'opblock')][1]"
    ]
    
    for selector in content_selectors:
        try:
            schema_content = driver.find_element(By.XPATH, selector)
            if schema_content.is_displayed() and schema_content.size['height'] > 50:
                print(f"Found schema content with selector: {selector}")
                break
        except:
            continue
    
    if not schema_content:
        # If still not found, use the schema element's parent container
        schema_content = driver.find_element(
            By.XPATH,
            "//button[contains(text(), 'Schema Definition')]/ancestor::div[3]"
        )

    # Ensure content is fully visible
    driver.execute_script("arguments[0].scrollIntoView({block: 'start'});", schema_content)
    time.sleep(1.5)

    # Zoom back to 100% before taking screenshot for better quality
    print("Zooming back to 100% for screenshot...")
    driver.execute_script("document.body.style.zoom='100%'")
    time.sleep(1)

    # Take screenshot
    print("Taking screenshot...")
    schema_content.screenshot("schema_definition.png")
    print("âœ“ Screenshot saved: schema_definition.png")

except TimeoutException as e:
    print(f"Timeout error: {e}")
    driver.save_screenshot("error_timeout.png")
    
except Exception as e:
    print(f"Error: {e}")
    driver.save_screenshot("error_general.png")
    print("Error screenshot saved for debugging")
    
finally:
    print("Closing browser...")
    time.sleep(2)
    driver.quit()
```

**Key changes:**

1. **Added zoom to 50%** using `document.body.style.zoom='50%'` after switching to iframe
2. **Wait for zoom to apply** (2 seconds)
3. **Zoom back to 100%** before taking the screenshot for better image quality
4. **Increased scroll increment** to 500px since content is zoomed out
5. **Reduced sleep time** between scrolls to 0.2s for faster execution

The zoom makes more content visible on screen, which should make the Schema Definition section findable. The screenshot is taken at 100% zoom for better quality.
